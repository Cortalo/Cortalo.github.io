$$

$$
udx + xdu = udx - (s/v) \sqrt{1+u^2} dx = udx - r\sqrt{1+u^2} dx
$$

$$
\dfrac{du}{\sqrt{1+u^2}} + r \dfrac{dx}{x} = 0
$$

$$
\ln \vert u + \sqrt{1+u^2} \vert + r \ln \vert x \vert = C
$$

$$
(u + \sqrt{1+u^2})x^r = k
$$

$$
(\dfrac{y}{x} + \sqrt{1+(\dfrac{y}{x})^2})x^r = k
$$

$$
